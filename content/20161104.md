#用for迴圈列出各組人員名單

content = open("cpa.txt").read()
#print(content)
# 這是單行註解
result = content.splitlines()
# result 變數資料形變為 list
#print(result)
# 針對 list 資料，可以利用 for 迴圈列出其數列成員
num = 1
for g in result:
    g_list = g.split(",")
    g_list = g_list[:-1]
    for m in g_list:
          print("group:", num, "member:", m)
    num = num + 1
